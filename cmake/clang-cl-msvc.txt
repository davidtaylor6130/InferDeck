# Toolchain file for clang-cl with MSVC compatibility

# Set clang-cl as the compiler BEFORE project()
set(CMAKE_C_COMPILER "clang-cl" CACHE STRING "C compiler" FORCE)
set(CMAKE_CXX_COMPILER "clang-cl" CACHE STRING "C++ compiler" FORCE)

# Use the MSVC standard library
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -fms-compatibility -fms-compatibility-version=1930")

# MSVC-like warnings
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} /W3 /WX- /utf-8")

# Disable some clang warnings that are noisy with MSVC headers
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wno-microsoft-enum-value -Wno-microsoft-anon-tag -Wno-microsoft-include")

# Use MSVC runtime
set(CMAKE_MSVC_RUNTIME_LIBRARY "MultiThreaded$<$<CONFIG:Debug>:Debug>")

# Link with MSVC libraries
set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} /MACHINE:X64")
set(CMAKE_SHARED_LINKER_FLAGS "${CMAKE_SHARED_LINKER_FLAGS} /MACHINE:X64")
set(CMAKE_MODULE_LINKER_FLAGS "${CMAKE_MODULE_LINKER_FLAGS} /MACHINE:X64")

# Windows SDK version
set(CMAKE_SYSTEM_VERSION "10.0")
